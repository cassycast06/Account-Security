<?php
// Start session
session_start();

// Check if form was submitted
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $username = htmlspecialchars($_POST['username']);
    $password = htmlspecialchars($_POST['password']);

    // Save login details to session (in real-world scenarios, store in a database)
    $_SESSION['logins'][] = [
        'username' => $username,
        'password' => $password
    ];

    // Redirect to a success page or back to the login
    header("Location: view_logins.php");
    exit();
}
?>
